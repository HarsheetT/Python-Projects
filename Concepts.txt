#Function_With_Multiple_Inputs

   Positional Arguments:
      def greet_with_name(name,location):
         print(f"Hello {name}")
         print(f"What is it like in {location}")
      
      greet_with_name("Jack","Mumbai")
   
   Positional Arguments:
      def greet_with_name(name,location):
         print(f"Hello {name}")
         print(f"What is it like in {location}")
      
      greet_with_name("Mumbai","Jack") ==> name = Mumbai,location=Jack

   Keyword Arguments
      def greet_with_name(name,location):
         print(f"Hello {name}")
         print(f"What is it like in {location}")
      
      greet_with_name(location = "Mumbai",name = "Jack") ==> name = Jack,location = Mumbai
